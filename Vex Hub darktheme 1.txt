            for _, player in pairs(game.Players:GetPlayers()) do

                            billboardGui = Instance.new("BillboardGui")

                            billboardGui.Size = UDim2.new(0, 200, 0, 50)

                            billboardGui.Parent = character

                            textLabel.Size = UDim2.new(0.5, 0, 0.5, 0)
                            textLabel.BackgroundTransparency = 1
                            textLabel.Text = player.Name
                            textLabel.TextColor3 = Color3.fromRGB(0, 255, 255)
                            textLabel.TextScaled = true
                            textLabel.Font = Enum.Font.SourceSansBold
                            textLabel.Parent = billboardGui
                        end
                    end
                end
            end
        end)
    else
        espEnabled = false
        if espConnection then
            espConnection:Disconnect()
        end
        for _, player in pairs(game.Players:GetPlayers()) do

local RunService = game:GetService("RunService")
local rotating = false
local spinSpeed = 1
local spinConnection

local Section = Tab:NewSection("Player manager")


local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- Selected Player Storage
local selectedPlayer = nil
local dropdown

local function updateDropdown()
    local playerNames = {}
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            table.insert(playerNames, player.Name)
        end
    end
    if dropdown then
        dropdown:Refresh(playerNames, false)
    end
end


dropdown = Section:NewDropdown("Select Player", "Choose a player", {}, function(option)
    selectedPlayer = Players:FindFirstChild(option)
end)

updateDropdown()

Players.PlayerAdded:Connect(updateDropdown)
Players.PlayerRemoving:Connect(updateDropdown)

Section:NewToggle("View/Unview", "Toggle to spectate selected player", function(state)
    if selectedPlayer and selectedPlayer.Character and selectedPlayer.Character:FindFirstChild("Head") then
        if state then
            workspace.CurrentCamera.CameraSubject = selectedPlayer.Character.Head
        else
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Head") then
                workspace.CurrentCamera.CameraSubject = LocalPlayer.Character.Head
            end
        end
    end
end)

Section:NewButton("Teleport", "Teleport to selected player", function()
    if selectedPlayer and selectedPlayer.Character and selectedPlayer.Character:FindFirstChild("HumanoidRootPart") then
        LocalPlayer.Character:MoveTo(selectedPlayer.Character.HumanoidRootPart.Position + Vector3.new(0, 3, 0))
    end
end)



local SpinSection = Tab:NewSection("Spin your Character")

SpinSection:NewToggle("Spin", "Spin your character", function(state)
    rotating = state
    if state then
        spinConnection = RunService.RenderStepped:Connect(function(dt)
            if hrp then
                local angle = math.rad(math.clamp(spinSpeed, 1, 100)) * dt * 60
                hrp.CFrame = hrp.CFrame * CFrame.Angles(0, angle, 0)
            end
        end)
    else
        if spinConnection then
            spinConnection:Disconnect()
            spinConnection = nil
        end
    end
end)

SpinSection:NewTextBox("Spin Velocity", "Insert velocity of the spin", function(txt)
    local num = tonumber(txt)
    if num then
        spinSpeed = math.clamp(num, 1, 100)
    end
end)

SpinSection:NewButton("Preset: Slow (5)", "Spin velocity: 5", function()
    spinSpeed = 10

end)

SpinSection:NewButton("Preset: fast (100)", "Spin velocity: 100", function()

end)

local Tab = Window:NewTab("Music Tab")
local Section = Tab:NewSection("NOT FE")
local Player = game.Players.LocalPlayer
local PlayerGui = Player:WaitForChild("PlayerGui")
local soundInstance

Section:NewTextBox("Sound ID", "Enter Roblox Sound ID (numbers only)", function(id)
    if soundInstance then
        soundInstance:Stop()
        soundInstance:Destroy()
        soundInstance = nil
    end

    if id:match("^%d+$") then
        soundInstance = Instance.new("Sound")
        soundInstance.SoundId = "rbxassetid://" .. id
        soundInstance.Volume = 1
        soundInstance.Looped = true
        soundInstance.Parent = PlayerGui
        soundInstance:Play()
        print("Trying to play sound:", soundInstance.SoundId)

        task.delay(1, function()
            if not soundInstance.Playing then
                warn("Sound failed to play. It may be private, deleted, or broken.")
            end
        end)
    else
        warn("Invalid sound ID! Please enter numbers only.")
    end
end)

Section:NewButton("Stop Music", "Stops the music", function()
    if soundInstance then
        soundInstance:Stop()
        soundInstance:Destroy()
        soundInstance = nil
    end
end)
